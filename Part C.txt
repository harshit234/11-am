Q1 — Conceptual

json.load() reads JSON data from a file object and converts it into a Python object (usually a dictionary or list).

Use json.load() when:

JSON data is stored in a file
You want to read the file and convert it to Python data

example - Many ML pipelines and APIs store config settings in JSON files.


json.loads() converts a JSON string into a Python object.

When to Use

Use json.loads() when:
JSON data is already in string format
Data is received from API responses
Data comes from message queues / logs

from pathlib import Path

def find_large_files(directory, size_mb):
    
    directory = Path(directory)
    results = []
    
    for file in directory.rglob("*"):
        
        if file.is_file():
            size_in_mb = file.stat().st_size / (1024 * 1024)
            
            if size_in_mb > size_mb:
                results.append((file.name, round(size_in_mb, 2)))
    
    results.sort(key=lambda x: x[1], reverse=True)
    
    return results

import csv

def merge_csv_files(file_list):
    
    all_data = []
    header_saved = False
    
    for filename in file_list:
        
        with open(filename, "r", newline="") as f:
            reader = csv.reader(f)
            header = next(reader)
            
            if not header_saved:
                all_data.append(header)
                header_saved = True
            
            for row in reader:
                all_data.append(row)
    
    with open("merged.csv", "w", newline="") as f:
        writer = csv.writer(f)
        writer.writerows(all_data)
    
    return len(all_data)
